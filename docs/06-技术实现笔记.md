## 2. Outliner 右键菜单扩展（方案C）

Blockbench 的 `OutlinerNode.showContextMenu(event)` 不提供 hook API，需要以下方式实现：

1. **进入模式时**：遍历所有 Group/Locator，保存原始 `showContextMenu` 方法引用
2. **替换方法**：用自定义函数包装，先调用原始方法获取菜单，再追加 MachineMax 专属项
3. **退出模式时**：恢复所有原始方法

伪代码：

```javascript
const originalMethods = new Map();

function patchContextMenus() {
    for (const group of Group.all) {
        const original = group.showContextMenu.bind(group);
        originalMethods.set(group, original);
        group.showContextMenu = function(event) {
            original(event);
            // 追加 MachineMax 菜单项
            addMMMenuItems(event);
        };
    }
}

function restoreContextMenus() {
    for (const [group, original] of originalMethods) {
        group.showContextMenu = original;
    }
    originalMethods.clear();
}
```

## 5. .bbmodel 持久化（Property 机制 + 双保险）

### 5.1 原理

Blockbench 的 `.bbmodel` compile/parse 遍历 `ModelProject.properties` 注册表（[bbmodel.js:L201-L204](file:///d:/Files/Project_MinecraftMods/blockbench/js/formats/bbmodel.js#L201-L204)）。注册为 Property 后自动纳入保存/加载，无需手动干预。

对 object 类型 Property，`copy()` 使用 `structuredClone`（深拷贝，property.ts:L217-219），`merge()` 使用 `structuredClone`（深拷贝恢复，property.ts:L189-191）。

### 5.2 实现代码

```javascript
// ===== utils/persistence.js =====

const CONFIG_VERSION = 3;
const PROPERTY_NAME = 'machine_max_plugin';

/**
 * 在插件 onload 时调用。注册 Property 到 ModelProject，
 * Blockbench 自动将 Project[PROPERTY_NAME] 纳入 .bbmodel 的保存/加载循环。
 */
export function registerProperty() {
    if (ModelProject.properties[PROPERTY_NAME]) {
        return; // 已注册（热重载场景）
    }
    new Property(ModelProject, 'object', PROPERTY_NAME, {
        default: {},
        exposed: false,
    });
}

/**
 * 从已恢复的 Project.machine_max_plugin 或备选文件加载配置。
 * 应在 Mode.onSelect 时调用。
 */
export function loadConfig(bbmodelPath) {
    // 1. 主方案：Property 已由 BB 自动恢复到内存
    if (Project[PROPERTY_NAME]?.$schema_version) {
        return migrateIfNeeded(Project[PROPERTY_NAME]);
    }

    // 2. 备选：独立文件保底
    const standalonePath = bbmodelPath.replace(/\.bbmodel$/, '.mm_project.json');
    if (fs.existsSync(standalonePath)) {
        try {
            const raw = JSON.parse(fs.readFileSync(standalonePath, 'utf-8'));
            const config = raw.config || raw;
            Project[PROPERTY_NAME] = migrateIfNeeded(config);
            return Project[PROPERTY_NAME];
        } catch (e) {
            console.warn('[MM Plugin] 备选配置文件读取失败:', e);
        }
    }

    // 3. 全新项目
    const blank = createBlankConfig();
    Project[PROPERTY_NAME] = blank;
    return blank;
}

/**
 * 保存配置。Property 自动纳入 Ctrl+S，这里额外写独立备份。
 * 应在 Mode.onUnselect 时调用，或监听 BB 保存事件。
 */
export function saveConfig(bbmodelPath, config) {
    // 主方案：直接赋值，Property 自动纳入下次 Ctrl+S
    Project[PROPERTY_NAME] = config;

    // 备选：独立文件保底
    const standalonePath = bbmodelPath.replace(/\.bbmodel$/, '.mm_project.json');
    try {
        fs.writeFileSync(
            standalonePath,
            JSON.stringify({
                $schema_version: CONFIG_VERSION,
                bbmodel: path.basename(bbmodelPath),
                timestamp: Date.now(),
                config: config,
            }, null, 2),
            'utf-8'
        );
    } catch (e) {
        console.warn('[MM Plugin] 备选配置写入失败:', e);
    }
}

function createBlankConfig() {
    return {
        $schema_version: CONFIG_VERSION,
        namespace: 'machine_max',
        parts: {},
        projectiles: {},
        connector_defs: {},
        subsystem_defs: {},
        material_defs: {},
    };
}
```

### 5.3 验证步骤

1. 在 BB 控制台执行：
   ```javascript
   new Property(ModelProject, 'object', 'machine_max_plugin', {default: {}, exposed: false});
   Project.machine_max_plugin = { $schema_version: 3, test: true, parts: {hello: {icon: "test"}} };
   ```
2. Ctrl+S 保存
3. 用文本编辑器打开 `.bbmodel`，搜索 `machine_max_plugin` — 应看到写入的 JSON
4. 关闭并重新在 BB 中打开该文件
5. 控制台输入 `Project.machine_max_plugin` — 应看到恢复的数据
6. 在**未加载插件**的情况下（或不注册 Property）打开该文件，数据仍在 JSON 中但不会被恢复到 `Project`。如果此时 Ctrl+S，Property 对应的 `machine_max_plugin` 字段将保留但子字段仍完好（因为没有被 copy）。

## 8. 拖入 JSON 导入

用户可将已有内容包 JSON 文件拖入 Blockbench 窗口进行反向解析回填。

```javascript
// ===== importers/part_importer.js =====

/**
 * 从 parts/*.json 反向解析并回填到 MMProjectConfig.parts
 */
export function importPartJSON(jsonPath, jsonContent) {
    const part = jsonContent;

    // 从文件名推断 partId
    const partId = path.basename(jsonPath, '.json');

    // 解析顶层字段
    const config = {
        icon: part.icon || '',
        vehicle_durability_rate: part.vehicle_durability_rate ?? 0.8,
        vehicle_damage_rate: part.vehicle_damage_rate ?? 1.0,
        vehicle_damage_rate_destroyed: part.vehicle_damage_rate_destroyed ?? 0.1,
        functional_threshold: part.functional_threshold ?? 0.3,
        share_durability: part.share_durability ?? true,
        max_stack_size: part.max_stack_size ?? 1,
        variants: {},
        element_markers: {},
    };

    // 解析变体（处理简写/完整格式）
    if (part.variants) {
        const variantsRaw = part.variants;
        if (variantsRaw.model) {
            // 简写格式：只有一个 default 变体
            config.variants['default'] = parseVariant(variantsRaw);
        } else {
            for (const [name, variant] of Object.entries(variantsRaw)) {
                config.variants[name] = parseVariant(variant);
            }
        }
    }

    MMProjectConfig.parts[partId] = config;
    return partId;
}

function parseVariant(variant) {
    const subParts = {};
    const subPartsRaw = variant.sub_parts;
    if (subPartsRaw) {
        if (subPartsRaw.start_bone !== undefined) {
            subParts['sub_part.machine_max.main'] = parseSubPart(subPartsRaw);
        } else {
            for (const [name, sp] of Object.entries(subPartsRaw)) {
                subParts[name] = parseSubPart(sp);
            }
        }
    }
    const textures = variant.textures;
    let parsedTextures = {};
    if (typeof textures === 'string' || Array.isArray(textures)) {
        parsedTextures = textures;
    } else if (typeof textures === 'object') {
        parsedTextures = textures;
    }
    return {
        model: variant.model || '',
        textures: parsedTextures,
        animations: variant.animations || null,
        tags: variant.tags || [],
        sub_parts: subParts,
    };
}

function parseSubPart(sp) {
    return {
        start_bone: sp.start_bone || '',
        end_bones: sp.end_bones || [],
        durability: sp.durability ?? 20.0,
        mass: sp.mass ?? 25.0,
        mass_center: sp.mass_center || 'mass_center',
        projected_area: sp.projected_area || [0, 0, 0],
        block_collision: sp.block_collision || 'true',
        collision_height: sp.collision_height ?? -1.0,
        climb_assist: sp.climb_assist ?? false,
        hydro_priority: sp.hydro_priority ?? 0,
        hit_boxes: sp.hit_boxes || {},
        interact_boxes: sp.interact_boxes || {},
        connectors: sp.connectors || {},
        subsystems: sp.subsystems || {},
        hydrodynamics: sp.hydrodynamics || null,
    };
}
```

**拖入拦截入口**（在 plugin.js onload 中注册）：

```javascript
// 监听文件拖入
document.addEventListener('dragover', (e) => e.preventDefault());
document.addEventListener('drop', (e) => {
    e.preventDefault();
    const files = e.dataTransfer.files;
    for (const file of files) {
        const filePath = file.path;
        if (filePath.includes('/parts/') || filePath.includes('\\parts\\')) {
            const content = JSON.parse(fs.readFileSync(filePath, 'utf-8'));
            importPartJSON(filePath, content);
            refreshUI();
        }
        // connectors/、subsystems/ 同理
    }
});
```
